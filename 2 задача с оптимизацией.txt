;R0=R1-R2-DataByte
MOV DataByte, R0 		;Записываем DataByte в R0
MOV R0, R1			;Копируем исходное число в R1
PUSH R0 			;Записываем исходное число из R0 в стек

;R0-A
AND 0007, R0 		;Логическая операция «И» с маской 0007 над R0

;R1-B
AND 0078, R1 		;Логическая операция «И» с маской 0078 над R1
SHR 3, R1 			;Логический сдвиг вправо R1 на 3 бита
;B-A
SUB R0,R1

POP R0 				;Возвращаем из стека исходное число в R0
;R2-C
AND 0080, R0 		;Логическая операция «И» с маской 0080 над R1
SHR 7, R0
;+C
ADD R0,R1

;RES
MOV R1, RES 			;Копируем результат в переменную RES

STOP
DataByte:
DATA D6
RES:
DATA 0
